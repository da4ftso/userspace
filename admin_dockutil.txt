# to-do: check for jamf, call policy; install pkg otherwise

if [ -e /usr/local/bin/dockutil ] ; then
	curl -O https://github.com/kcrawford/dockutil/releases/download/3.0.2/dockutil-3.0.2.pkg ;
	installer -pkg dockutil-3.0.2.pkg -target / ;
	sleep 5 ;
	echo "dockutil installed.."
fi

/usr/local/bin/dockutil --remove all --no-restart ~ ;
sleep 5 ;
